#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <pthread.h>
pthread_mutex_t mutex = PTHREAD_MUTEX_INITIALIZER;
void lire()
{
    printf("lire debut\n");
    usleep(rand() % 500000);
    printf("lire fin\n");
}

void ecrire()
{
    printf("ecrire debut\n");
    usleep(rand() % 500000);
    printf("ecrire fin\n");
}

void *run(void *arg)
{
    while (1)
    {
        if ((rand() % 2) == 0)
        {
            pthread_mutex_lock(&mutex);
            lire();
            pthread_mutex_unlock(&mutex);
        }
        else
        {
            pthread_mutex_lock(&mutex);
            ecrire();
            pthread_mutex_unlock(&mutex);
        }
    }
}

int main()
{
    int N = 4;
        pthread_t T[N];
    for (int i = 0; i < N; i++)
        pthread_create(&T[i], NULL, run, NULL);
    for (int i = 0; i < N; i++)
        pthread_join(T[i], NULL);
    return 0;
}

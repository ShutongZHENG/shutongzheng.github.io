Exo1

给出下列表达式的类型

fun x -> x + 1 

fun x y -> x *. y  

fun x (y, z) -> if x then y else not z

fun x -> x 

fun x y z -> (x <= y, z)

Correction:

int -> int

float -> float -> float

bool -> (bool * bool) -> bool

'a -> 'a

'a -> 'a -> 'b -> (bool * 'b)

Exo4

给出尾递归(recursive terminale)版本

let rec h x y =
    if x = y then 1
    else
        if x < y then x + y + h (x - 1) (y - 2)
        else x * h (x-1) y ;;

Correction:

let h x y =
    let rec h1 x y acc = 
    if x = y then acc
    else 
        if x < y then h1 (x-1) (y-2) (x+y+acc)
        else h1 (x-1) y (x*acc)
in h1 x y 1 ;;
